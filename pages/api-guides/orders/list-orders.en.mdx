## List Open Orders by Maker

<Tabs items={['GraphQL', 'Python', 'Rust', 'Typescript']}>
  <Tab>
    <MakerAddress>
    ```graphql copy
    {
    offers( where: { open : true, maker: "0xinput_address" } , first: 1000) {
        # the asset the offer is selling 
        pay_gem
        # the amount of the pay_gem that is being sold (not converted)
        pay_amt
        # the amount of the offer that has been paid out (not converted)
        paid_amt
        # the asset the offer is buying 
        buy_gem
        # the amount of the buy_gem that is being sold (not converted)
        buy_amt
        # the amount of the offer that has been bought (not converted)
        bought_amt
        # the offer owner (the address that controls the offer)
        maker { id }
      }
    }
    ```
    </MakerAddress>

    import requests
    import json

    url = "select network endpoint below"

    headers = {'Content-Type': 'application/json'}

    query = """
    {
    offers(where: { open: true, maker: "0xinput_address" }, first: 1000) {
        pay_gem
        pay_amt
        paid_amt
        buy_gem
        buy_amt
        bought_amt
        maker { id }
      }
    }
    """

    response = requests.post(url, headers=headers, data=json.dumps({'query': query}))

    data = response.json()

    print(data)

    ```rust copy
    use reqwest::Error;
    use serde_json::{json, Value};

    #[tokio::main]
    async fn main() -> Result<(), Error> {
        let client = reqwest::Client::new();
        let query = r#"
        {
            offers(where: { open: true, maker: "0xinput_address" }, first: 1000) {
                pay_gem
                pay_amt
                paid_amt
                buy_gem
                buy_amt
                bought_amt
                maker { id }
            }
        }
        "#;

        let payload = json!({
            "query": query
        });

        let url = "select network endpoint below";
        let res: Value = client.post(url)
            .json(&payload)
            .send()
            .await?
            .json()
            .await?;

        println!("{:#?}", res);

        Ok(())
    }
    ```
